Last.fm based on cui.jsonp

How to use:


cui.ready(function(){

         new LastFM({username: 'olivboy', where: 'container',
                           onRequest: function(data){
                                $('#container').innerHTML = "Loading...";
                           }
         }).get();

         new LastFM({username: 'voidberg', where: 'container2', 
                            onRequest: function(data){
                                $('#container2').innerHTML = "Loading...";
                            }
         }).get();    
});
